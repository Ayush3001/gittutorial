git status-to check the current status 	
git init-to initialise a repository
git add --a : to add all the files in staging area
git commit -m "message" : to make a commit with message
git log: to check the commits you have made
git add file_name : to add a specific file
rm -rf .git : to delete the repository
git clone url : clone the repository
 *.extension : add it in the gitignore to ignore all the files that have specific extension
 git diff : tells the difference between staging area and working directory
git diff --staged : compares the staging area with the previous commit
git commit -a -m "message" : directly commits the tracked files
git rm filename: deletes the specific file
git mv filename new filename : rename the file
